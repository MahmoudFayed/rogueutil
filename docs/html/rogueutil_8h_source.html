105 int
106 kbhit(void)
107 {
108  static struct termios oldt, newt;
109  int cnt = 0;
110  tcgetattr(STDIN_FILENO, &oldt);
111  newt = oldt;
112  newt.c_lflag &= ~(ICANON | ECHO);
113  newt.c_iflag = 0; /* input mode */
114  newt.c_oflag = 0; /* output mode */
115  newt.c_cc[VMIN] = 1; /* minimum time to wait */
116  newt.c_cc[VTIME] = 1; /* minimum characters to wait for */
117  tcsetattr(STDIN_FILENO, TCSANOW, &newt);
118  ioctl(0, FIONREAD, &cnt); /* Read count */
119  struct timeval tv;
120  tv.tv_sec = 0;
121  tv.tv_usec = 100;
122  select(STDIN_FILENO+1, NULL, NULL, NULL, &tv); /* A small time delay */
123  tcsetattr(STDIN_FILENO, TCSANOW, &oldt);
124  return cnt; /* Return number of characters */
125 }
